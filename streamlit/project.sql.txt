-- 5. STORED PROCEDURE

CREATE OR REPLACE PROCEDURE ANALYTICS.STORE_PROCEDURE_PIPELINE()
RETURNS STRING
LANGUAGE SQL
AS
$$
DECLARE
    v_rows_inserted NUMBER DEFAULT 0;
BEGIN

    INSERT INTO ANALYTICS.FACT_SALES (
        INVOICE_NO,
        STOCK_CODE,
        DESCRIPTION,
        QUANTITY,
        INVOICE_TS,
        INVOICE_DATE,
        UNIT_PRICE,
        CUSTOMER_ID,
        COUNTRY,
        LINE_AMOUNT,
        TRANSACTION_TYPE,
        LOAD_TS
    )
    SELECT
        INVOICE_NO,
        STOCK_CODE,
        DESCRIPTION,
        QUANTITY,
        INVOICE_TS,
        INVOICE_DATE,
        UNIT_PRICE,
        CUSTOMER_ID,
        COUNTRY,
        QUANTITY * UNIT_PRICE,
        CASE
            WHEN INVOICE_NO ILIKE 'C%' THEN 'CANCELLATION'
            WHEN QUANTITY < 0 THEN 'RETURN'
            ELSE 'SALE'
        END,
        CURRENT_TIMESTAMP()
    FROM CLEAN.CLEAN_ECOMMERCE_DATA;

    v_rows_inserted := SQLROWCOUNT;

    CREATE OR REPLACE TABLE ANALYTICS.SALES_TREND AS
    SELECT
        INVOICE_DATE,
        COUNT(DISTINCT INVOICE_NO) AS TOTAL_ORDER,
        SUM(QUANTITY) AS TOTAL_ITEM_SOLD,
        SUM(LINE_AMOUNT) AS TOTAL_REVENUE
    FROM ANALYTICS.FACT_SALES
    WHERE TRANSACTION_TYPE = 'SALE'
    GROUP BY INVOICE_DATE;

    CREATE OR REPLACE TABLE ANALYTICS.PRODUCT_PERFORMANCE AS
    SELECT
        STOCK_CODE,
        DESCRIPTION,
        SUM(CASE WHEN TRANSACTION_TYPE = 'SALE' THEN QUANTITY ELSE 0 END) AS TOTAL_QUANTITY_SOLD,
        SUM(CASE WHEN TRANSACTION_TYPE = 'SALE' THEN LINE_AMOUNT ELSE 0 END) AS TOTAL_REVENUE,
        COUNT(DISTINCT CASE WHEN TRANSACTION_TYPE = 'SALE' THEN INVOICE_NO END) AS TOTAL_ORDERS
    FROM ANALYTICS.FACT_SALES
    GROUP BY STOCK_CODE, DESCRIPTION;

    CREATE OR REPLACE TABLE ANALYTICS.CUSTOMER_BEHAVIOR AS
    SELECT
        CUSTOMER_ID,
        COUNT(DISTINCT CASE WHEN TRANSACTION_TYPE = 'SALE' THEN INVOICE_NO END) AS TOTAL_ORDER,
        SUM(CASE WHEN TRANSACTION_TYPE = 'SALE' THEN LINE_AMOUNT ELSE 0 END) AS TOTAL_SPENT,
        AVG(CASE WHEN TRANSACTION_TYPE = 'SALE' THEN LINE_AMOUNT END) AS AVG_ORDER_VALUE,
        MIN(CASE WHEN TRANSACTION_TYPE = 'SALE' THEN INVOICE_DATE END) AS FIRST_ORDER_DATE,
        MAX(CASE WHEN TRANSACTION_TYPE = 'SALE' THEN INVOICE_DATE END) AS LAST_ORDER_DATE
    FROM ANALYTICS.FACT_SALES
    WHERE CUSTOMER_ID IS NOT NULL
    GROUP BY CUSTOMER_ID;

    CREATE OR REPLACE TABLE ANALYTICS.CUSTOMER_SEGMENTATION AS
    SELECT
        CUSTOMER_ID,
        TOTAL_ORDER,
        TOTAL_SPENT,
        AVG_ORDER_VALUE,
        FIRST_ORDER_DATE,
        LAST_ORDER_DATE,
        CASE
            WHEN TOTAL_SPENT >= 2000 THEN 'HIGH VALUE'
            WHEN TOTAL_SPENT >= 1000 THEN 'MEDIUM VALUE'
            ELSE 'LOW VALUE'
        END AS CUSTOMER_SEGMENT
    FROM ANALYTICS.CUSTOMER_BEHAVIOR;

    CREATE OR REPLACE TABLE ANALYTICS.RETURN_ANALYTICS AS
    SELECT
        STOCK_CODE,
        DESCRIPTION,
        COUNT(*) AS RETURN_ROW,
        SUM(ABS(QUANTITY)) AS TOTAL_ITEM_RETURNED,
        SUM(ABS(LINE_AMOUNT)) AS RETURN_VALUE
    FROM ANALYTICS.FACT_SALES
    WHERE TRANSACTION_TYPE = 'RETURN'
      AND DESCRIPTION NOT ILIKE '%fee%'
      AND DESCRIPTION NOT ILIKE '%charge%'
      AND DESCRIPTION NOT ILIKE '%discount%'
      AND DESCRIPTION NOT ILIKE '%manual%'
    GROUP BY STOCK_CODE, DESCRIPTION;

    CREATE OR REPLACE TABLE ANALYTICS.NET_SALES AS
    SELECT
        INVOICE_DATE,
        SUM(CASE WHEN TRANSACTION_TYPE = 'SALE' THEN LINE_AMOUNT ELSE 0 END) AS GROSS_SALES,
        SUM(CASE WHEN TRANSACTION_TYPE IN ('RETURN', 'CANCELLATION') THEN ABS(LINE_AMOUNT) ELSE 0 END) AS RETURNED_VALUE,
        SUM(
            CASE
                WHEN TRANSACTION_TYPE = 'SALE' THEN LINE_AMOUNT
                WHEN TRANSACTION_TYPE IN ('RETURN', 'CANCELLATION') THEN -ABS(LINE_AMOUNT)
                ELSE 0
            END
        ) AS NET_SALES
    FROM ANALYTICS.FACT_SALES
    GROUP BY INVOICE_DATE;

    INSERT INTO CONTROL_SCHEMA.PIPELINE_RUN_LOGG
    VALUES (
        CURRENT_TIMESTAMP(),
        'ANALYTICS.STORE_PROCEDURE_PIPELINE',
        :v_rows_inserted,
        'SUCCESS',
        'Fact and analytics tables refreshed successfully'
    );

    RETURN 'SUCCESS - Rows inserted into FACT_SALES: ' || v_rows_inserted;

EXCEPTION
    WHEN OTHER THEN
        INSERT INTO CONTROL_SCHEMA.PIPELINE_RUN_LOGG
        VALUES (
            CURRENT_TIMESTAMP(),
            'ANALYTICS.STORE_PROCEDURE_PIPELINE',
            0,
            'FAILED',
            :SQLERRM
        );

        RETURN 'FAILED - ' || :SQLERRM;
END;
$$;

-- 6. TASK
CREATE OR REPLACE TASK ANALYTICS.TASK_RUN_PIPELINE
WAREHOUSE = COMPUTE_WH
SCHEDULE = 'USING CRON 0 * * * * UTC'
AS
CALL ANALYTICS.STORE_PROCEDURE_PIPELINE();

ALTER TASK ANALYTICS.TASK_RUN_PIPELINE RESUME;

-- 8. DYNAMIC TABLES
CREATE OR REPLACE DYNAMIC TABLE ANALYTICS.DT_SALES_DB
TARGET_LAG = '5 MINUTES'
WAREHOUSE = 'COMPUTE_WH'
AS
SELECT
    st.INVOICE_DATE,
    st.TOTAL_ORDER,
    st.TOTAL_ITEM_SOLD,
    st.TOTAL_REVENUE,
    ns.GROSS_SALES,
    ns.RETURNED_VALUE,
    ns.NET_SALES
FROM ANALYTICS.SALES_TREND st
JOIN ANALYTICS.NET_SALES ns
    ON st.INVOICE_DATE = ns.INVOICE_DATE;

CREATE OR REPLACE DYNAMIC TABLE ANALYTICS.DT_PRODUCT_DB
TARGET_LAG = '5 MINUTES'
WAREHOUSE = 'COMPUTE_WH'
AS
SELECT
    pp.STOCK_CODE,
    pp.DESCRIPTION,
    pp.TOTAL_QUANTITY_SOLD,
    pp.TOTAL_REVENUE,
    pp.TOTAL_ORDERS,
    COALESCE(ra.RETURN_ROW, 0) AS RETURN_ROW,
    COALESCE(ra.TOTAL_ITEM_RETURNED, 0) AS TOTAL_ITEM_RETURNED,
    COALESCE(ra.RETURN_VALUE, 0) AS RETURN_VALUE
FROM ANALYTICS.PRODUCT_PERFORMANCE pp
LEFT JOIN ANALYTICS.RETURN_ANALYTICS ra
    ON pp.STOCK_CODE = ra.STOCK_CODE
   AND pp.DESCRIPTION = ra.DESCRIPTION;

CREATE OR REPLACE DYNAMIC TABLE ANALYTICS.DT_CUSTOMER_DB
TARGET_LAG = '5 MINUTES'
WAREHOUSE = 'COMPUTE_WH'
AS
SELECT
    cb.CUSTOMER_ID,
    cb.TOTAL_ORDER,
    cb.TOTAL_SPENT,
    cb.AVG_ORDER_VALUE,
    cb.FIRST_ORDER_DATE,
    cb.LAST_ORDER_DATE,
    cs.CUSTOMER_SEGMENT
FROM ANALYTICS.CUSTOMER_BEHAVIOR cb
JOIN ANALYTICS.CUSTOMER_SEGMENTATION cs
    ON cb.CUSTOMER_ID = cs.CUSTOMER_ID;
